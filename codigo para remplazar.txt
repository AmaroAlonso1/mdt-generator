    <script>
        // Establecer la fecha de la computadora al abrir
        window.onload = function () {
            const hoy = new Date();
            const año = hoy.getFullYear();
            const mes = String(hoy.getMonth() + 1).padStart(2, '0');
            const dia = String(hoy.getDate()).padStart(2, '0');
            document.getElementById('fecha').value = `${año}-${mes}-${dia}`;
            actualizarFormulario();
        };

        // ------------- FUNCIONES DE VALIDACIÓN -------------
        function validarLetras(input) {
            input.value = input.value.replace(/[^a-zA-ZáéíóúÁÉÍÓÚñÑ\s]/g, '');
        }

        function validarNumbers(input) {
            input.value = input.value.replace(/[^0-9]/g, '');
        }

        function validarSujetos(input) {
            input.value = input.value.replace(/[^0-9]/g, '');
            if (input.value !== '') {
                let valor = parseInt(input.value);
                if (valor > 20) {
                    input.value = 20;
                }
            }
        }

        // ------------- LÓGICA DE INTERFAZ DINÁMICA -------------
        function actualizarFormulario() {
            let tipo = document.getElementById('tipo_procedimiento').value;
            let grupoCheckbox = document.getElementById('grupo_checkbox_negociacion');
            let labelDelito = document.getElementById('label_delito');

            if (tipo === 'robo') {
                grupoCheckbox.style.display = 'block';
                labelDelito.innerText = 'Robo a Establecimiento:';
            } else {
                grupoCheckbox.style.display = 'none';
                labelDelito.innerText = 'Acción de ZR:';
                document.getElementById('hubo_negociacion').checked = false;
            }
            actualizarNegociacion();
        }

        function actualizarNegociacion() {
            let tipo = document.getElementById('tipo_procedimiento').value;
            let chkNegociacion = document.getElementById('hubo_negociacion').checked;
            let grupoNegociacion = document.getElementById('grupo_negociacion');

            if (tipo === 'robo' && chkNegociacion) {
                grupoNegociacion.style.display = 'block';
            } else {
                grupoNegociacion.style.display = 'none';
            }
        }

        // ------------- GENERADOR DE BITÁCORA -------------
        function generarBitacora() {
            let fechaInput = document.getElementById('fecha').value;
            let fechaPartes = fechaInput.split('-');
            let fechaFormateada = `${fechaPartes[2]}/${fechaPartes[1]}/${fechaPartes[0]}`;

            let casillero = document.getElementById('casillero').value;
            let tipo = document.getElementById('tipo_procedimiento').value;
            let delito = document.getElementById('delito').value;
            let chkNegociacion = document.getElementById('hubo_negociacion').checked;
            let negociacion = document.getElementById('negociacion').value;
            let cantidad = document.getElementById('cantidad').value || "1";
            let calibre = document.getElementById('calibre').value;
            let vestimenta = document.getElementById('vestimenta').value;
            let vehiculo = document.getElementById('vehiculo').value;
            let colorVehiculo = document.getElementById('color_vehiculo').value;
            let cp = document.getElementById('codigo_postal').value;

            // Añadir palabra "Robo a " si aplica
            let delitoFormateado = (tipo === 'robo') ? `Robo a ${delito}` : delito;

            let texto = `Fecha: ${fechaFormateada}\nBCSO involucrados: \nNro de casillero: ${casillero}\n\nResumen de procedimiento:\n\n`;

            let relato = `\`Llegaría un llamado sobre ${cantidad} sujeto/s, quienes estarían realizando ${delitoFormateado}, estos vestirían de ${vestimenta}, portarían ${calibre} calibre y se transportarían en ${vehiculo} color ${colorVehiculo}, Mando ordenaría a los oficiales acudir de manera inmediata a la zona. `;

            if (tipo === 'robo') {
                if (chkNegociacion) {
                    relato += `una vez llegado, se procedería a entablar negociaciones, se cerrarían las negociaciones en: ${negociacion}, tras varios minutos de procedimiento, este terminaría con un 10-99 en las cercanías de la postal ${cp}, al haber abatido a la totalidad de sujetos. `;
                } else {
                    relato += `una vez llegado, comenzaría un tiroteo el cual, tras varios minutos de procedimiento, terminaría con un 10-99 en las cercanías de la postal ${cp}, al haber abatido a la totalidad de sujetos. `;
                }
            } else if (tipo === 'zona_roja') {
                relato += `una vez llegado, los oficiales se posicionarían de manera estratégica para hacer ingreso a la zona en las cercanías de ${cp}, al ingresar comenzaría un tiroteo que terminaría con un 10-99 al acabar con todos los sujetos abatidos. `;
            }

            relato += `se solicitaría 10-38 en lo que EMS acudiría en helicóptero a la zona, reanimaría a los sujetos, se les leería sus derechos y serían trasladados para cumplir su condena\位`;

            // Cierre
            relato += `se solicitaría 10-38 en lo que EMS acudiría en helicóptero a la zona, reanimaría a los sujetos, se les leería sus derechos y serían trasladados para cumplir su condena\``;

            document.getElementById('resultado').value = texto + relato;
        }

        function copiarTexto() {
            let texto = document.getElementById('resultado');
            texto.select();
            document.execCommand("copy");
        }
    </script>
